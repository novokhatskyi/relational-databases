-- Результат відсортуйте за порахованим середнім значенням у порядку спадання.
-- Оберіть тільки 10 рядків для виведення на екран.

select count(*) as null_count
from infectious_cases ic 
where TRIM(Number_rabies) = '';

select 
	e.entity,
	e.code,
	round(min(ic.Number_rabies), 2) as min_rabies,
	round(avg(ic.Number_rabies), 2) as mean_rabies,
	round(max(ic.Number_rabies), 2) as max_rabies,
	round(SUM(ic.Number_rabies), 2) as sum_rabies
from infectious_cases ic 
JOIN entities e
on e.id = ic.entity_id 
group by e.entity, e.code
order by mean_rabies desc
limit 10

-- функція має приймати на вхід значення року, а повертати різницю в роках між поточною датою та датою, 
-- створеною з атрибута року (1996 рік → ‘1996-01-01’).

delimiter //

create function CalculateYearDiff (year_infection int)
returns INT
deterministic
begin
	RETURN TIMESTAMPDIFF(YEAR, MAKEDATE(year_infection, 1), CURDATE());
end //

delimiter ;
